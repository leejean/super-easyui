	function tipsHover(o) {
	
	if (o.e.type == "mouseover") {
		var top = o.this.offset().top + o.top,
			left = o.this.offset().left + o.left,
		    width = o.width,
		    height =o.height,
			div = $('<div class=tips-hover>' + o.divObj[0].outerHTML + '</div>');
			div.css({
				position: 'absolute',
				zIndex: 1000,
				background:o.bgColor,
				top: top + 'px',
				left: left + 'px',
		    	width:width+'px',
		    	height:height+'px'
			});
			o.divObj.removeClass('dn');
			div.insertAfter(o.this);
		}
		if (o.e.type == "mouseout") {
			$(".tips-hover").remove();
		}
	}
	
	$(document).on('mouseover mouseout', 'a', function(e) {
		tipsHover({
			this: $(this),
			e: e,
			divObj: $('<div style="width:100px;height:20px;">物流信息<p>到达北京</p></div>'),
			bgColor:'#fff',
			width:200,
			height:100,
			top: 20,
			left: 30
		});
	});
